name: Kuro kainų surinkimas

on:
  schedule:
    # 10:30 Lietuvos laiku (UTC+2 žiemą, UTC+3 vasarą)
    # Vasarą (EEST): 10:30 LT = 07:30 UTC
    # Žiemą (EET):   10:30 LT = 08:30 UTC
    # Naudojame 07:30 UTC (vasaros laikas kovo-spalio mėn.)
    - cron: '30 7 * * 1-5'   # Darbo dienomis 07:30 UTC (10:30 EEST)

  workflow_dispatch:  # Rankinis paleidimas

jobs:
  collect-prices:
    runs-on: ubuntu-latest
    timeout-minutes: 30

    steps:
      - name: Checkout repozitorija
        uses: actions/checkout@v4

      - name: Python setup
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Įdiegti priklausomybes
        run: |
          pip install -r requirements.txt
          playwright install chromium
          playwright install-deps

      - name: Sukurti cookies direktoriją
        run: mkdir -p cookies debug

      - name: Paleisti kainų surinkimą
        env:
          CK_ADBLUE_URL: ${{ secrets.CK_ADBLUE_URL }}
          NESTE_URL: ${{ secrets.NESTE_URL }}
          NESTE_EMAIL: ${{ secrets.NESTE_EMAIL }}
          NESTE_PASSWORD: ${{ secrets.NESTE_PASSWORD }}
          AS24_URL: ${{ secrets.AS24_URL }}
          AS24_EMAIL: ${{ secrets.AS24_EMAIL }}
          AS24_PASSWORD: ${{ secrets.AS24_PASSWORD }}
          AS24_CLIENT_ID: ${{ secrets.AS24_CLIENT_ID }}
        run: python main.py

      - name: Commit ir push JSON
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add kuro_kainos.json
          git diff --cached --quiet || git commit -m "Atnaujintos kuro kainos $(date +%Y-%m-%d)"
          git push

      - name: Įkelti debug screenshots (jei yra)
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: debug-screenshots
          path: debug/
          retention-days: 7
